def getHist(channel):
  unique, counts = np.unique(channel, return_counts=True)
  points=dict(zip(unique, counts))
  for i in range(256):
    if i not in points:
      points[i]=0
  hist=[]
  for i in range(256):
    hist.append(points[i])
  return np.array(hist)
